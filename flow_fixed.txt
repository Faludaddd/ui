local Players = game:GetService('Players')
local UIS = game:GetService('UserInputService')
local TweenService = game:GetService('TweenService')
local CoreGui = game:GetService('CoreGui')
local LocalPlayer = Players.LocalPlayer

pcall(function()
    local old = CoreGui:FindFirstChild('Flow') or LocalPlayer.PlayerGui:FindFirstChild('Flow')
    if old then old:Destroy() end
end)

local Flow = {}
Flow.Themes = {
    Mono = {
        Background = Color3.fromRGB(10, 11, 15),
        Sidebar = Color3.fromRGB(14, 16, 22),
        Topbar = Color3.fromRGB(15, 17, 24),
        Card = Color3.fromRGB(18, 20, 28),
        Stroke = Color3.fromRGB(45, 48, 60),
        Text = Color3.fromRGB(245, 245, 245),
        SubText = Color3.fromRGB(155, 160, 175),
        Accent = Color3.fromRGB(255, 255, 255)
    }
}

local function protect(gui)
    local ok = pcall(function() gui.Parent = CoreGui end)
    if not ok then gui.Parent = LocalPlayer:WaitForChild('PlayerGui') end
    return gui
end

local function inst(class, props, children)
    local o = Instance.new(class)
    for k,v in pairs(props or {}) do o[k] = v end
    for _,c in ipairs(children or {}) do c.Parent = o end
    return o
end

local function corner(r) return inst('UICorner', {CornerRadius = UDim.new(0, r or 8)}) end
local function stroke(col, tr) return inst('UIStroke', {Color = col, Transparency = tr or 0.65, Thickness = 1}) end
local function pad(l,t,r,b) return inst('UIPadding', {PaddingLeft=UDim.new(0,l or 0),PaddingTop=UDim.new(0,t or 0),PaddingRight=UDim.new(0,r or 0),PaddingBottom=UDim.new(0,b or 0)}) end

function Flow.New(opts)
    opts = opts or {}
    local theme = opts.Theme or Flow.Themes.Mono
    local app = {Theme = theme}

    local gui = inst('ScreenGui', {
        Name='Flow', IgnoreGuiInset=true, ResetOnSpawn=false, ZIndexBehavior=Enum.ZIndexBehavior.Sibling, DisplayOrder=999999
    })
    protect(gui)
    app.Gui = gui

    function app:Window(wopts)
        wopts = wopts or {}
        local win = {Tabs={}, Theme=theme}
        local size = wopts.Size or UDim2.fromOffset(860,520)

        local main = inst('Frame', {
            Name='Window', AnchorPoint=Vector2.new(.5,.5), Position=UDim2.fromScale(.5,.5), Size=size,
            BackgroundColor3=theme.Background, BorderSizePixel=0, Active=true
        }, {corner(10), stroke(theme.Stroke, .45)})
        main.Parent = gui
        win.Root = main

        local top = inst('Frame', {Name='Topbar', Size=UDim2.new(1,0,0,56), BackgroundColor3=theme.Topbar, BorderSizePixel=0}, {corner(10), stroke(theme.Stroke,.75)})
        top.Parent = main
        local title = inst('TextLabel', {BackgroundTransparency=1, Position=UDim2.fromOffset(70,0), Size=UDim2.new(0,240,1,0), Font=Enum.Font.GothamBold, Text=wopts.Title or 'Flow UI', TextColor3=theme.Text, TextSize=16, TextXAlignment=Enum.TextXAlignment.Left})
        title.Parent = top
        local sideBtn = inst('TextButton', {BackgroundTransparency=1, Position=UDim2.fromOffset(18,16), Size=UDim2.fromOffset(24,24), Text='☰', TextColor3=theme.Text, TextSize=18, Font=Enum.Font.GothamBold})
        sideBtn.Parent = top
        local search = inst('TextBox', {BackgroundColor3=Color3.fromRGB(12,13,18), BackgroundTransparency=.15, Position=UDim2.new(1,-230,0,13), Size=UDim2.fromOffset(190,30), PlaceholderText='Search', Text='', TextColor3=theme.Text, PlaceholderColor3=theme.SubText, TextSize=13, Font=Enum.Font.Gotham, BorderSizePixel=0, ClearTextOnFocus=false})
        search.Parent = top; corner(7).Parent = search; stroke(theme.Stroke,.7).Parent = search
        local close = inst('TextButton', {BackgroundTransparency=1, Position=UDim2.new(1,-34,0,14), Size=UDim2.fromOffset(26,26), Text='×', Font=Enum.Font.GothamBold, TextSize=18, TextColor3=theme.SubText})
        close.Parent = top; close.MouseButton1Click:Connect(function() gui:Destroy() end)

        local sidebar = inst('Frame', {Name='Sidebar', Position=UDim2.fromOffset(0,56), Size=UDim2.new(0,190,1,-56), BackgroundColor3=theme.Sidebar, BorderSizePixel=0}, {pad(12,14,12,12)})
        sidebar.Parent = main
        local sideList = inst('UIListLayout', {Padding=UDim.new(0,8), SortOrder=Enum.SortOrder.LayoutOrder})
        sideList.Parent = sidebar

        local content = inst('Frame', {Name='Content', Position=UDim2.fromOffset(190,56), Size=UDim2.new(1,-190,1,-56), BackgroundTransparency=1, BorderSizePixel=0})
        content.Parent = main

        do -- drag
            local dragging, dragStart, startPos
            top.InputBegan:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                    dragging = true; dragStart = input.Position; startPos = main.Position
                    input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
                end
            end)
            UIS.InputChanged:Connect(function(input)
                if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                    local d = input.Position - dragStart
                    main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + d.X, startPos.Y.Scale, startPos.Y.Offset + d.Y)
                end
            end)
        end

        local sidebarOpen = true
        sideBtn.MouseButton1Click:Connect(function()
            sidebarOpen = not sidebarOpen
            TweenService:Create(sidebar, TweenInfo.new(.25), {Size = sidebarOpen and UDim2.new(0,190,1,-56) or UDim2.new(0,0,1,-56)}):Play()
            TweenService:Create(content, TweenInfo.new(.25), {Position = sidebarOpen and UDim2.fromOffset(190,56) or UDim2.fromOffset(0,56), Size = sidebarOpen and UDim2.new(1,-190,1,-56) or UDim2.new(1,0,1,-56)}):Play()
        end)

        function win:Tab(topts)
            topts = topts or {}
            local tab = {Window=win, Items={}}
            local btn = inst('TextButton', {Name='Tab', Size=UDim2.new(1,0,0,34), BackgroundColor3=Color3.fromRGB(255,255,255), BackgroundTransparency=.93, Text=topts.Title or 'Tab', TextColor3=theme.SubText, TextSize=13, Font=Enum.Font.GothamMedium, BorderSizePixel=0, AutoButtonColor=false, TextXAlignment=Enum.TextXAlignment.Left}, {corner(7), pad(12,0,8,0)})
            btn.Parent = sidebar
            local page = inst('ScrollingFrame', {Name=(topts.Title or 'Tab')..'Page', BackgroundTransparency=1, BorderSizePixel=0, Size=UDim2.fromScale(1,1), CanvasSize=UDim2.new(), AutomaticCanvasSize=Enum.AutomaticSize.Y, ScrollBarThickness=3, Visible=false}, {pad(18,18,18,18)})
            page.Parent = content
            inst('UIListLayout', {Padding=UDim.new(0,10), SortOrder=Enum.SortOrder.LayoutOrder}).Parent = page
            tab.Button=btn; tab.Page=page
            table.insert(win.Tabs, tab)
            local function select()
                for _,t in ipairs(win.Tabs) do
                    t.Page.Visible=false; t.Button.TextColor3=theme.SubText; t.Button.BackgroundTransparency=.93
                end
                page.Visible=true; btn.TextColor3=theme.Text; btn.BackgroundTransparency=.84
            end
            btn.MouseButton1Click:Connect(select)
            if topts.Selected or #win.Tabs == 1 then select() end

            local function addCard(height)
                local card = inst('Frame', {Size=UDim2.new(1,0,0,height or 42), BackgroundColor3=theme.Card, BorderSizePixel=0}, {corner(8), stroke(theme.Stroke,.65)})
                card.Parent = page
                return card
            end
            function tab:Section(sopts)
                local lbl = inst('TextLabel', {BackgroundTransparency=1, Size=UDim2.new(1,0,0,24), Text=sopts.Title or 'Section', TextColor3=theme.Text, TextSize=14, Font=Enum.Font.GothamBold, TextXAlignment=Enum.TextXAlignment.Left})
                lbl.Parent = page
                return lbl
            end
            function tab:Label(lopts)
                local card = addCard(34)
                inst('TextLabel', {BackgroundTransparency=1, Position=UDim2.fromOffset(12,0), Size=UDim2.new(1,-24,1,0), Text=lopts.Text or lopts.Title or 'Label', TextColor3=theme.SubText, TextSize=13, Font=Enum.Font.Gotham, TextXAlignment=Enum.TextXAlignment.Left}).Parent = card
                return card
            end
            function tab:Button(bopts)
                local card = addCard(42)
                local b = inst('TextButton', {Position=UDim2.fromOffset(8,7), Size=UDim2.new(1,-16,1,-14), BackgroundColor3=theme.Accent, Text=bopts.Label or bopts.Title or 'Button', TextColor3=Color3.fromRGB(0,0,0), TextSize=13, Font=Enum.Font.GothamBold, BorderSizePixel=0, AutoButtonColor=false}, {corner(7)})
                b.Parent=card
                b.MouseButton1Click:Connect(function() if bopts.Pushed then bopts.Pushed() elseif bopts.Callback then bopts.Callback() end end)
                return b
            end
            function tab:Toggle(tg)
                local card = addCard(44)
                inst('TextLabel', {BackgroundTransparency=1, Position=UDim2.fromOffset(12,0), Size=UDim2.new(1,-70,1,0), Text=tg.Title or 'Toggle', TextColor3=theme.Text, TextSize=13, Font=Enum.Font.GothamMedium, TextXAlignment=Enum.TextXAlignment.Left}).Parent = card
                local on = tg.Value == true
                local sw = inst('TextButton', {Position=UDim2.new(1,-54,.5,-11), Size=UDim2.fromOffset(42,22), BackgroundColor3= on and theme.Accent or Color3.fromRGB(40,43,52), Text='', BorderSizePixel=0, AutoButtonColor=false}, {corner(20)})
                sw.Parent=card
                local knob = inst('Frame', {Position= on and UDim2.fromOffset(22,3) or UDim2.fromOffset(3,3), Size=UDim2.fromOffset(16,16), BackgroundColor3= on and Color3.fromRGB(0,0,0) or theme.SubText, BorderSizePixel=0}, {corner(20)})
                knob.Parent=sw
                local function set(v)
                    on=v; TweenService:Create(sw,TweenInfo.new(.18),{BackgroundColor3=on and theme.Accent or Color3.fromRGB(40,43,52)}):Play(); TweenService:Create(knob,TweenInfo.new(.18),{Position=on and UDim2.fromOffset(22,3) or UDim2.fromOffset(3,3), BackgroundColor3=on and Color3.fromRGB(0,0,0) or theme.SubText}):Play()
                    if tg.ValueChanged then tg.ValueChanged(nil,on) elseif tg.Callback then tg.Callback(on) end
                end
                sw.MouseButton1Click:Connect(function() set(not on) end)
                return {Set=set, Get=function() return on end}
            end
            function tab:Slider(sl)
                local min,max,val = sl.Minimum or sl.Min or 0, sl.Maximum or sl.Max or 100, sl.Value or 0
                local card = addCard(58)
                local label = inst('TextLabel', {BackgroundTransparency=1, Position=UDim2.fromOffset(12,4), Size=UDim2.new(1,-24,0,22), Text=(sl.Title or 'Slider')..'  '..tostring(val), TextColor3=theme.Text, TextSize=13, Font=Enum.Font.GothamMedium, TextXAlignment=Enum.TextXAlignment.Left})
                label.Parent=card
                local bar = inst('Frame', {Position=UDim2.fromOffset(12,36), Size=UDim2.new(1,-24,0,6), BackgroundColor3=Color3.fromRGB(40,43,52), BorderSizePixel=0}, {corner(10)})
                bar.Parent=card
                local fill = inst('Frame', {Size=UDim2.new((val-min)/(max-min),0,1,0), BackgroundColor3=theme.Accent, BorderSizePixel=0}, {corner(10)})
                fill.Parent=bar
                local dragging=false
                local function setFromX(x)
                    local pct=math.clamp((x-bar.AbsolutePosition.X)/bar.AbsoluteSize.X,0,1)
                    val=math.floor((min+(max-min)*pct)*100)/100
                    fill.Size=UDim2.new(pct,0,1,0); label.Text=(sl.Title or 'Slider')..'  '..tostring(val)
                    if sl.ValueChanged then sl.ValueChanged(nil,val) elseif sl.Callback then sl.Callback(val) end
                end
                bar.InputBegan:Connect(function(i) if i.UserInputType==Enum.UserInputType.MouseButton1 or i.UserInputType==Enum.UserInputType.Touch then dragging=true; setFromX(i.Position.X) end end)
                UIS.InputChanged:Connect(function(i) if dragging and (i.UserInputType==Enum.UserInputType.MouseMovement or i.UserInputType==Enum.UserInputType.Touch) then setFromX(i.Position.X) end end)
                UIS.InputEnded:Connect(function(i) if i.UserInputType==Enum.UserInputType.MouseButton1 or i.UserInputType==Enum.UserInputType.Touch then dragging=false end end)
                return {Set=function(v) val=math.clamp(v,min,max); local pct=(val-min)/(max-min); fill.Size=UDim2.new(pct,0,1,0); label.Text=(sl.Title or 'Slider')..'  '..tostring(val) end, Get=function() return val end}
            end
            return tab
        end
        return win
    end
    return app
end

-- Demo, remove this part after you confirm it works
local app = Flow.New({Theme = Flow.Themes.Mono})
local win = app:Window({Title='Flow UI', Size=UDim2.fromOffset(860,520)})
local main = win:Tab({Title='Main', Selected=true})
main:Section({Title='Showcase'})
main:Toggle({Title='Auto Farm', Value=false, ValueChanged=function(_,v) print('Auto Farm:',v) end})
main:Slider({Title='Walk Speed', Minimum=16, Maximum=150, Value=16, ValueChanged=function(_,v)
    local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass('Humanoid')
    if hum then hum.WalkSpeed = v end
end})
main:Button({Label='Test Button', Pushed=function() print('Flow button clicked') end})
local settings = win:Tab({Title='Settings'})
settings:Section({Title='UI'})
settings:Label({Text='This is the fixed one-file Flow UI.'})
settings:Button({Label='Destroy UI', Pushed=function() gui:Destroy() end})

return Flow
